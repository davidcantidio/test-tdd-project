        raise ValueError("Data cannot be None")
    
    if not isinstance(data, dict):
        raise TypeError(f"Expected dict, got {type(data)}")
    
    if "value" not in data:
        raise KeyError("Required key 'value' not found in data")
    
    return data["value"]
